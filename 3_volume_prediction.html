select

        when cast(strftime('%m', date) as integer) between 4 and 6 then strftime('%Y', date) || ' Q2'

        else strftime('%Y', date) || ' Q4' end as YQ
    , sum(bet_euro) as total_bet_euro
from FACT_BETS
group by YQ
order by date(date);
"""
